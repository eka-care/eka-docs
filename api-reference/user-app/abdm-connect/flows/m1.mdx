      C1 --> C1R["Resend OTP"]

      C3 -->|Yes| C4["Create"]
      C3 -->|No| C5["Mobile Verify OTP"]
      C5 --> C5R["Mobile Resend OTP"]
      C5 --> C6["Create"]
    end

    subgraph CF1["Face Auth"]
      direction TB
      F1["Face Auth Init"] --> F2["Capture Face"]
      F2 --> F3["Verify"]
      F2 -.-> FP["Polling API"]
    end

    subgraph CM1["Mobile"]
      direction TB
      M1["Generate OTP"] --> M2["Verify OTP"]
      M1 --> M1R["Resend OTP"]
      M2 --> M3["Create"]
    end

    C2 -.-> F3
  end

  subgraph LOGIN["Login ABHA"]
    direction LR

    subgraph LP1["phr_address"]
      direction TB
      P1["Generate OTP"] --> P2["Verify OTP"]

    end

    subgraph LA1["aadhaar"]
      direction TB
      L1["Generate OTP"] --> L2["Verify OTP"]

      L4 --> L5["skip_state = abha_end"]

      LM3 --> LM4["PHR Login API"]

  end

  CREATE --- N1
  LOGIN --- N2
  LOGIN --- N3
